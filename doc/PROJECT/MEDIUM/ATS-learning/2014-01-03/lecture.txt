######

constants:

0, 1, 2, 100, ~1, ~10, 'a', 'b',
"Zoe", "Chloe", "Hello, world!", ...

######

names:

x, y, z, foo, bar, baz, ...

######

expressions:

1+2, x*100-3*5, x*x + y*y, 1/x

######

declarations:

val x = 1 + 2 // introducing name x
val y = x * 5 // introducing name y
val z = y - x // introducing name z

fun foo (x) = x + 1

fun foo2 (x, y) = x * y + 1

fun fact (n) = if n > 0 then n * fact(n-1) else 1
